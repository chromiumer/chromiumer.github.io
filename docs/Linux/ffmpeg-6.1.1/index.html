1.nasm
tar -xvf nasm-2.14.02.tar.xz && cd nasm-2.14.02

2.x264
tar -xvf x264-master.tar.bz2 && cd x264-master
./configure --enable-shared && make -j8 && make install && cd ../
\cp /usr/local/lib/pkgconfig/x264.pc /usr/share/pkgconfig/

3.ffmpeg
tar -xvf ffmpeg-6.1.1.tar.xz && cd ffmpeg-6.1.1
./configure --prefix=/usr/local/ffmpeg --enable-gpl --enable-libx264 --enable-version3 --enable-openssl && make -j8 && make install && cd ../
