数据量统计​

/*库统计-所有数据总量*/
select SUM((DATA_LENGTH+INDEX_LENGTH)/1024/1024) as `sum_data（MB）`
from information_schema.tables
where table_schema not in ('information_schema','performance_schema','mysql','sys'); 

/*库统计-按库*/
select * from (
select TABLE_SCHEMA as "schema",sum(DATA_LENGTH/1024/1024) as `data（MB）`,sum(INDEX_LENGTH/1024/1024) as `index（MB）`,sum((DATA_LENGTH+INDEX_LENGTH)/1024/1024) as `sum（MB）`,now() as `curtime`
from information_schema.tables
where TABLE_SCHEMA not in ('information_schema','performance_schema','mysql','sys')
group by TABLE_SCHEMA)  as t
order by `sum（MB）` desc;


/*表统计-按库*/
select  table_schema as 'schema',table_name as 'TableName',(DATA_LENGTH/1024/1024) as `data(MB)`,(INDEX_LENGTH/1024/1024) as `index(MB)`,((DATA_LENGTH+INDEX_LENGTH)/1024/1024) as `data+index(MB)`,table_rows as 'records'
from          information_schema.tables
where         table_schema = 'ccx_pro'
order   by  records desc;

/*表统计-所有库*/
select  table_schema as 'schema',table_name as 'table_name',(DATA_LENGTH/1024/1024) as `data(MB)`,(INDEX_LENGTH/1024/1024) as `index(MB)`,((DATA_LENGTH+INDEX_LENGTH)/1024/1024) as `data+index(MB)`,table_rows as 'records'
from          information_schema.tables
where TABLE_SCHEMA not in ('information_schema','performance_schema','mysql','sys')
order   by  records desc;

SELECT
	TABLE_SCHEMA,
	TABLE_NAME,
	TABLE_TYPE,
	ifnull( ENGINE, 'NONE' ) AS ENGINE,
	/* /ifnull( VERSION, '0' ) AS VERSION, */
	ifnull( ROW_FORMAT, 'NONE' ) AS ROW_FORMAT,
	ifnull( TABLE_ROWS, '0' ) AS TABLE_ROWS,
	ifnull( DATA_LENGTH, '0' ) AS DATA_LENGTH,
	ifnull( INDEX_LENGTH, '0' ) AS INDEX_LENGTH,
	ifnull( DATA_FREE, '0' ) AS DATA_FREE,
	ifnull( CREATE_OPTIONS, 'NONE' ) AS CREATE_OPTIONS,
	TABLE_COMMENT
FROM
	information_schema.TABLES 
WHERE
	TABLE_SCHEMA = 'hap_job_clsh';

/*表统计--业务基表信息*/
select TABLE_SCHEMA,TABLE_NAME,TABLE_TYPE,ENGINE,TABLE_ROWS,DATA_LENGTH,INDEX_LENGTH,DATA_FREE,CREATE_TIME,UPDATE_TIME,TABLE_COLLATION,TABLE_COMMENT from TABLES where TABLE_SCHEMA not in ("information_schema","performance_schema","mysql","sys","test");

/*表统计--业务表数量*/
select table_schema,count(1) as tables_count from information_schema.tables where table_schema not in ('information_schema','performance_schema','mysql','sys','test') group by table_schema order by count(1) desc;

select count(1),DB from information_schema.processlist group by DB order by count(*) desc;

/*会话连接-按用户*/
select count(1),user from information_schema.processlist group by user order by count(*) desc;
 
/*会话连接-非Sleep状态*/
select * from  information_schema.processlist where Command <> "Sleep";

/*会话连接-非Sleep状态超200秒*/
select * from  information_schema.processlist where Command <> "Sleep" and time > 200;

/*生成kill cmd*/
select concat("kill(",id,");")  from  information_schema.processlist where Command <> "Sleep" and time > 20;
 
/*会话连接-按db和host统计连接数量*/
select count(1),db,substring_index(host,':','1') host from information_schema.processlist group by substring_index(host,':','1'),db order by count(*) desc;
